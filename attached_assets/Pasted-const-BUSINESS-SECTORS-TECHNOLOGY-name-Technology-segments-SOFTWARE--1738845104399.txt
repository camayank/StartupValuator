// Utility functions for sector and segment operations
const sectorOperations = {
  getAllSectors() {
    return Object.keys(BUSINESS_SECTORS).map(key => ({
      key,
      name: BUSINESS_SECTORS[key].name
    }));
  },

  getSegmentsForSector(sectorKey) {
    const sector = BUSINESS_SECTORS[sectorKey];
    return sector ? Object.keys(sector.segments).map(key => ({
      key,
      name: sector.segments[key].name
    })) : [];
  },

  getSubSegments(sectorKey, segmentKey) {
    const sector = BUSINESS_SECTORS[sectorKey];
    const segment = sector?.segments[segmentKey];
    return segment ? segment.subSegments : [];
  },

  getMetrics(sectorKey, segmentKey) {
    const sector = BUSINESS_SECTORS[sectorKey];
    const segment = sector?.segments[segmentKey];
    return segment ? segment.metrics : null;
  }
};

export { BUSINESS_SECTORS, sectorOperations };
