const IndustrySpecificValidation = {
  SaaS: {
    revenueValidation: (data) => {
      const { mrr, userBase, pricePoint } = data;
      const expectedRevenue = userBase * pricePoint;
      
      return {
        isValid: Math.abs(mrr - expectedRevenue) < expectedRevenue * 0.2,
        warning: 'MRR seems inconsistent with user base and pricing'
      };
    },

    metrics: {
      required: ['mrr', 'arr', 'churnRate', 'cac', 'ltv'],
      recommended: ['expansionRevenue', 'nps', 'activeUsers']
    }
  },

  Ecommerce: {
    revenueValidation: (data) => {
      const { gmv, aov, orderCount } = data;
      const expectedGMV = aov * orderCount;
      
      return {
        isValid: Math.abs(gmv - expectedGMV) < expectedGMV * 0.1,
        warning: 'GMV inconsistent with orders and AOV'
      };
    },

    metrics: {
      required: ['gmv', 'aov', 'orderCount', 'returnRate'],
      recommended: ['repeatPurchaseRate', 'inventoryTurnover']
    }
  }
};
